\node (decision5) [decision, left=1cm of decision1] {$T^p_j \leq T^{cv}_j$?};

\node (pro2) [process, below=1cm of decision1] {Set $T^s_j = T^p_j$};

\node (decision4) [decision, right=1cm of decision1] {$T^s_j < T^{nv}_j$?};

\node (decision3) [decision, below=1cm of pro3] {Is Gleason score greater than 6?};

\node (stop) [startstop, below = 1cm of decision3] {Remove patient from AS};
